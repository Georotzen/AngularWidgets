Support for pui-disabled

        
        

        
        
						
	$scope.showMessage = function(msg) {
		alert(msg);
	};

	$scope.buttonDisabled = true;

	$scope.enableButton = function() {
		$scope.buttonDisabled = false;
	};

	$scope.disableButton = function() {
		$scope.buttonDisabled =  true;
	};
